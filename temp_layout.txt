---
import Header from '../components/common/Header.astro';
import Footer from '../components/common/Footer.astro';
import { getTranslations, type Lang } from '../i18n/ui';
import { getLangFromUrl } from '../i18n/utils';
import '../styles/global.css';

interface Props {
  title: string;
  description: string;
  ogImage?: string;
  articleSchema?: object;
}

const { title, description, ogImage = '/og-default.jpg', articleSchema } = Astro.props;
const lang = getLangFromUrl(Astro.url);
const t = getTranslations(lang);

const locales: Lang[] = ['en', 'zh-cn', 'ja', 'ko', 'es', 'pt'];
const currentPath = Astro.url.pathname.replace(/^\/(en|zh-cn|ja|ko|es|pt)/, '');
---

<!doctype html>
<html lang={lang}>
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <meta name="description" content={description} />
    <meta name="generator" content={Astro.generator} />
    <meta name="theme-color" content="#dc2626" />

    <!-- PWA -->
    <link rel="manifest" href="/manifest.json" />

    <!-- Open Graph -->
    <meta property="og:title" content={title} />
    <meta property="og:description" content={description} />
    <meta property="og:image" content={ogImage} />
    <meta property="og:type" content="website" />
    <meta property="og:locale" content={lang} />

    <!-- Twitter Card -->
    <meta name="twitter:card" content="summary_large_image" />
    <meta name="twitter:title" content={title} />
    <meta name="twitter:description" content={description} />

    <!-- Canonical -->
    <link rel="canonical" href={Astro.url.href} />

    <!-- hreflang alternates -->
    {locales.map((loc) => (
      <link rel="alternate" hreflang={loc} href={`/${loc}${currentPath}`} />
    ))}
    <link rel="alternate" hreflang="x-default" href={`/en${currentPath}`} />

    <!-- RSS -->
    <link rel="alternate" type="application/rss+xml" title={`Crimson Guide RSS (${lang})`} href={`/${lang}/rss.xml`} />

    <!-- Fonts -->
    <link rel="preconnect" href="https://fonts.googleapis.com" />
    <link rel="preconnect" href="https://fonts.gstatic.com" crossorigin />
    <link
      href="https://fonts.googleapis.com/css2?family=Inter:wght@400;500;600;700;800&display=swap"
      rel="stylesheet"
    />

    <!-- Favicon -->
    <link rel="icon" type="image/svg+xml" href="/favicon.svg" />

    <title>{title} | Crimson Guide</title>

    <!-- JSON-LD Structured Data -->
    {articleSchema ? (
      <script type="application/ld+json" set:html={JSON.stringify(articleSchema)} />
    ) : (
      <script type="application/ld+json" set:html={JSON.stringify({
        "@context": "https://schema.org",
        "@type": "WebSite",
        "name": "Crimson Guide",
        "url": "https://crimsonsguide.com",
        "description": description,
        "inLanguage": locales,
      })} />
    )}
  </head>
  <body class="min-h-screen flex flex-col bg-white text-surface-900 dark:bg-surface-950 dark:text-surface-100">
    <Header lang={lang} />
    <main class="flex-1">
      <slot />
    </main>
    <Footer lang={lang} />

    <script is:inline>
      // Theme init â€?runs before render to avoid flash
      const theme = localStorage.getItem('theme') || (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light');
      document.documentElement.classList.toggle('dark', theme === 'dark');
    </script>
  </body>
</html>
